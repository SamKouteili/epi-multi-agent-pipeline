                if url.endswith(".csv") or "csv" in url:
                    from io import StringIO
                    cholera_df_raw = pd.read_csv(StringIO(resp.text))
                    print(f"  CSV columns: {cholera_df_raw.columns.tolist()[:10]}")
                    cholera_df = cholera_df_raw
                    break
                elif url.endswith("CHOLERA_0000000001"):
                    data = resp.json()
                    print(f"  JSON keys: {list(data.keys())[:5]}")
                    records = data.get("value", [])
                    if records:
                        df_temp = pd.DataFrame(records)
                        print(f"  JSON columns: {df_temp.columns.tolist()[:10]}")
                        cholera_df = df_temp
                        break
            else:
                print(f"  HTTP {resp.status_code}")
        except Exception as e:

# ---------------------------------------------------------------------------
# Process cholera data if acquired

print("\nStep 3: Processing acquired data...")

data_acquired = False

if cholera_df is not None and len(cholera_df) > 10:
    print(f"  Cholera data shape: {cholera_df.shape}")

    # Attempt to standardize columns to iso, year, proxy_value
    try:

            cholera_clean["proxy_value"] = pd.to_numeric(cholera_clean["proxy_value"], errors="coerce")
            cholera_clean = cholera_clean.dropna()
            print(f"  Cleaned cholera data: {len(cholera_clean)} rows")
            
            target = load_raw_indicator("UWD")

            merged = target.merge(cholera_clean, on=["iso", "year"])

            if len(merged) >= 20:

        else:

    # Partial correlation controlling for log(GPC)
    gpc = load_raw_indicator("GPC")
    merged_gpc = merged.merge(

    print(f"  Best functional form: {form.best_form.value}")

    verdict = determine_verdict(corr, partial, "positive")
    print(f"  Verdict: {verdict.value}")

    result = build_result_json(

        data_quality_notes=(

        summary=(

    result["verification_method"] = "statistical_test"

else:

    print("  Citation: Lowe et al. 2017, Emerging Infectious Diseases (CDC peer-reviewed journal)")

    # Load UWD to get basic stats
