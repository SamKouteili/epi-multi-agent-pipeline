            print(f"    Top results:\n{results[['id','name']].head(3).to_string()}")
    except Exception as e:

]

proxy_name = None
proxy_wb_code = None
proxy_col = "proxy_value"

    try:

            print(f"    Got {len(df)} rows, {df['iso'].nunique()} countries")
            merged_test = target.merge(df, on=["iso", "year"])
